```python
@router.get("/admin/platform-health")
async def get_platform_health(request: Request):
    """
    Six seed-milestone KPIs for the admin dashboard and investor demos.
    MAU, DAU, active venues, active merchants, total ratings, estimated MRR.
    Super-admin only.
    """
    user = await get_current_user(request)
    if not user or not user.get("is_super_admin"):
        raise HTTPException(status_code=403, detail="Super-admin only")

    now = datetime.now(timezone.utc)
    thirty_days_ago = now - timedelta(days=30)
    twenty_four_hours_ago = now - timedelta(hours=24)

    # MAU — distinct users who submitted a rating in the last 30 days
    mau_pipeline = [
        {"$match": {"timestamp": {"$gte": thirty_days_ago}}},
        {"$group": {"_id": "$user_id"}},
        {"$count": "total"},
    ]
    mau_result = await db.ratings.aggregate(mau_pipeline).to_list(1)
    mau = mau_result[0]["total"] if mau_result else 0

    # DAU — distinct users who submitted a rating in the last 24 hours
    dau_pipeline = [
        {"$match": {"timestamp": {"$gte": twenty_four_hours_ago}}},
        {"$group": {"_id": "$user_id"}},
        {"$count": "total"},
    ]
    dau_result = await db.ratings.aggregate(dau_pipeline).to_list(1)
    dau = dau_result[0]["total"] if dau_result else 0

    # Active venues — venues with is_active=true (or verified=true as fallback)
    active_venues = await db.venues.count_documents(
        {"$or": [{"is_active": True}, {"verified": True}]}
    )

    # Active merchants — merchant_subscriptions with status=active
    # Fall back to counting distinct venue owners if collection is empty
    active_merchants = await db.merchant_subscriptions.count_documents({"status": "active"})
    if active_merchants == 0:
        # Fallback: count venues that have a merchant_wallet (means a merchant exists)
        active_merchants = await db.merchant_wallets.count_documents({})

    # Total ratings submitted (all time)
    total_ratings = await db.ratings.count_documents({})

    # Estimated MRR (NGN)
    active_vibe_plus = await db.subscriptions.count_documents(
        {"status": "active", "plan": "vibe_plus"}
    )
    estimated_mrr = (active_vibe_plus * 2000) + (active_merchants * 150000)

    return {
        "mau": mau,
        "dau": dau,
        "active_venues": active_venues,
        "active_merchants": active_merchants,
        "total_ratings": total_ratings,
        "estimated_mrr_ngn": estimated_mrr,
        "computed_at": now.isoformat(),
    }
```
